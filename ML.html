Let A be a matrix of size:(n,k)

And B be a matrix of size:(k,m)

Then multiplication matrix C=A×B would be of size:(n,m). For each element in C, we have to do k multiplication and (k-1) addition, hence

Total number of multiplication=n×m×k

Total number of addition=n×m×(k-1)

import time
def multiplication(A,B):
  n=len(A)
  k=len(A[0])
  m=len(B[0])

  C=[[0]*m for _ in range(n)]
  for i in range(n):
    for j in range(m):
      C[i][j]=sum(A[i][l]*B[l][j] for l in range(k))
  return C

B=[[6,8],[19,10],[1,2]]

import numpy as np
import time

def median(arr):
    sor=sorted(arr)
    n=len(sor)
    mid=n//2
    if n%2==0:
        return (sor[mid-1]+sor[mid])/2
    else:
        return sor[mid]

def median_quickselect(arr):
    def quickselect(arr, k):
        if len(arr)==1:
            return arr[0]

        pivot=arr[len(arr) // 2]
        lows=[el for el in arr if el < pivot]
        highs=[el for el in arr if el > pivot]
        pivots=[el for el in arr if el == pivot]

        if k < len(lows):
            return quickselect(lows, k)
        elif k < len(lows) + len(pivots):
            return pivots[0]
        else:
            return quickselect(highs, k - len(lows) - len(pivots))

    n=len(arr)
    if n%2==1:
        return quickselect(arr, n // 2)
    else:
        return (quickselect(arr, n // 2 - 1) + quickselect(arr, n // 2)) / 2

np.random.seed(0)
arr=np.random.randint(1, 1000, size=1000).tolist()

s=time.time()
Sort_result=median(arr)
tim=time.time()-s

s=time.time()
quickselect_result=median_quickselect(arr)
tim2=time.time()-s

# Timing numpy.median
s=time.time()
numpy_result=np.median(arr)
tim3=time.time()-s

print("Median using sorting:",Sort_result,"Time:",tim)
print("Median using Quickselect:",quickselect_result,"Time:",tim2)
print("Median using numpy.median:", numpy_result,"Time:",tim3)

$$\frac{\partial f}{\partial x}= 2xy+ y^3cosx$$ Partial derivative of f wrt y: $$\frac{\partial f}{\partial y}= x^2+ 3y^2 sinx$$ Hence $$ \nabla f=(2xy+ y^3cosx,x^2+ 3y^2 sinx)$$

    return 2*x*y + y**3 * jnp.cos(x)

def del_fy(x, y):
    return x**2 + 3*(y**2) * jnp.sin(x)

import numpy as np

def find(arr):
    Mi=0
    for i in range(1,len(arr)):
        if arr[i]<arr[Mi]:
            Mi=i
    return Mi
